## sort bedfiles

# gawk if statement
gawk -v OFS="\t" '{if ($6 == "+") print $1,$2,$2,$4,$5,$6; else print $1, $3, $3, $4,$5,$6}' hg19_ucsc_genes_sorted_s.bed  > hg19_tss.bed
